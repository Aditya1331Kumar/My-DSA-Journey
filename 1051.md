10. two array compare kaise kare in o(n) way
```

class Solution {
    public int heightChecker(int[] heights) {
        int[] arr = new int[101];
        for(int x:heights) arr[x]++;

        int res=0;
        int j=0;
        for(int i=0;i<101;i++){
            while(arr[i]>0){
                if(heights[j++]!=i) res++;
                arr[i]--;
            }
        }
        return res;
    }
}
```
