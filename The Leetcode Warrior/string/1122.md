import java.util.Arrays;

class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        int n = arr1.length;
        int m = 0; // next position to place element that matches arr2

        for (int i = 0; i < arr2.length; i++) {
            int x = arr2[i];
            for (int j = m; j < n; j++) {
                if (arr1[j] == x) {
                    swap(arr1, m, j); // move match to position m
                    m++;              // only now increase m
                }
            }
        }

        // elements from index m to n-1 are not in arr2 => sort them
        Arrays.sort(arr1, m, n);
        return arr1;
    }

    static void swap(int[] arr, int a, int b) {
        int t = arr[a];
        arr[a] = arr[b];
        arr[b] = t;
    }
}

import java.util.*;

class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        int maxVal = 1000; // problem constraint often <= 1000
        int[] freq = new int[maxVal + 1];
        for (int v : arr1) freq[v]++;

        int idx = 0;
        // place elements in arr2 order
        for (int v : arr2) {
            while (freq[v]-- > 0) arr1[idx++] = v;
        }
        // place remaining sorted
        for (int v = 0; v <= maxVal; v++) {
            while (freq[v]-- > 0) arr1[idx++] = v;
        }
        return arr1;
    }
}

class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        HashMap<Integer,Integer> hm = new HashMap<>();
        for (int j : arr1) {
            if (hm.containsKey(j)) {
                hm.put(j, hm.get(j) + 1);
            } else {
                hm.put(j, 1);
            }
        }
        int[] ans = new int[arr1.length];
        int cnt=0;
        for (int j : arr2) {
            while (hm.get(j) > 0) {
                ans[cnt] = j;
                cnt++;
                hm.put(j, hm.get(j) - 1);
            }
            hm.remove(j);
        }
        int[] remaining = new int[arr1.length-cnt];
        int index = 0;
        for (int j : arr1) {
            if (hm.containsKey(j)) {
                while (hm.get(j) > 0) {
                    remaining[index] = j;
                    index++;
                    hm.put(j, hm.get(j) - 1);
                }
                hm.remove(j);
            }
        }
        Arrays.sort(remaining);
        for (int j : remaining) {
            ans[cnt] = j;
            cnt++;
        }
        return ans;
    }
}
