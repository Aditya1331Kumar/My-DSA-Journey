class Solution {
   public static int[] shuffle (int[] nums, int n) {
         int[] ans = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {

            if (i % 2 == 0) {
                ans[i]=nums[(i+1)/2];
            }
            else {
                ans[i] = nums[(n+i/2)];
            }
        }
        return ans;
    }
}

class Solution {
    public int[] shuffle(int[] nums, int n) {
        int[] ans = new int[2*n];
        int index = 0;

        for(int i = 0; i < n; i++){
            ans[index++] = nums[i];
            ans[index++] = nums[n + i];
        }
        return ans;
    }
}
