class Solution {
    public List<Integer> selfDividingNumbers(int left, int right) {
        List<Integer> ans = new ArrayList<>();
        for (int num = left; num <= right; num++) {
            int temp = num;
            boolean ok = true;

            while (temp > 0) {
                int digit = temp % 10;
                if (digit == 0 || (num % digit) != 0) { // check 0 first
                    ok = false;
                    break;
                }
                temp /= 10;
            }

            if (ok) ans.add(num);
        }
        return ans;
    }
}

class Solution {
    public static boolean div(int a,int b)
    {
        int z = b%10;
        if(z==0) return false;
        if(b<10) return a%z==0;
        return (a%z==0) && div(a,b/10);
    }
    public List<Integer> selfDividingNumbers(int left, int right) {
        List<Integer> l = new ArrayList<>();
        for(int i=left;i<=right;i++)
        {
            if(div(i,i)) l.add(i);
        }
        return l;
        
    }
}

public int floor(int[] arr, int target) {
    int start = 0;
    int end = arr.length - 1;

    while (start <= end) {
        int mid = start + (end - start) / 2;

        if (target > arr[mid]) { 
            end = mid - 1;   // yahan > hoga kyunki descending hai
        } else if (target < arr[mid]) {
            start = mid + 1;
        } else {
            return mid;
        }
    }
    return start; // descending me start floor ke equal hoga
}
