1.   class Solution {
    public String sortSentence(String s) {
        String[] arr = s.split(" ");
        int i = 0;

        while (i < arr.length) {
            int l = arr[i].length();
            int correctIndex = (arr[i].charAt(l - 1) - '0') - 1; // position ‚Üí 0-based index

            if (correctIndex != i) {
                swap(arr, i, correctIndex);
            } else {
                i++;
            }
        }

        // Remove trailing digits and join
        for (int k = 0; k < arr.length; k++) {
            arr[k] = arr[k].substring(0, arr[k].length() - 1);
        }
        return String.join(" ", arr);
    }

    private void swap(String[] arr, int first, int second) {
        String temp = arr[first];
        arr[first] = arr[second];
        arr[second] = temp;
    }
}

5.class Solution {
    public String sortSentence(String s) {
        String[] arr = s.split(" ");       // 1Ô∏è‚É£
        String[] res = new String[arr.length]; // 2Ô∏è‚É£

        for (String element : arr) {       // 3Ô∏è‚É£
            int l = element.length();
            
            // üîç Find where digits start from end
            int j = l - 1;  
            while (j >= 0 && Character.isDigit(element.charAt(j))) j--;  // 4Ô∏è‚É£
            
            // üéØ Extract position number
            int pos = Integer.parseInt(element.substring(j + 1)); // 5Ô∏è‚É£
            
            // ‚úÇ Remove the digits from the word
            String word = element.substring(0, j + 1);            // 6Ô∏è‚É£
            
            // üìå Place the word in correct position
            res[pos - 1] = word;                                  // 7Ô∏è‚É£
        }

        return String.join(" ", res);     // 8Ô∏è‚É£
    }
}
